How to simply make a dynimic UITableViewHeader animation

2020/3/30 posted in  iOS

1.add a imageView to a headerView
2.set headerView as UITableView`s tableHeaderView
3.observer scrollViewDidScroll Method



class Demo: UITableViewController {

    let header = UIView(frame: CGRect(x: 0, y: 0, width: screenWidth, height: headerViewH))
    let imageV = UIImageView(frame: CGRect(x: 0, y: 0, width: screenWidth, height: headerViewH))

    override func viewDidLoad() {
        super.viewDidLoad()
        header.backgroundColor = .white
        header.addSubview(imageV)
        tableView.contentInsetAdjustmentBehavior = .never
        imageV.image = UIImage(named: "logo")
        imageV.contentMode = .scaleAspectFit
        imageV.topAnchor.constraint(equalTo: header.topAnchor).isActive = true
        imageV.bottomAnchor.constraint(equalTo: header.bottomAnchor).isActive = true
        imageV.leadingAnchor.constraint(equalTo: header.leadingAnchor).isActive = true
        imageV.bottomAnchor.constraint(equalTo: header.bottomAnchor).isActive = true
        tableView.register(UITableViewCell.self, forCellReuseIdentifier: "cell")
        tableView.tableHeaderView = header
   
    }
    

....

   override func scrollViewDidScroll(_ scrollView: UIScrollView) {
        // 1. get scrollView`s y-axis offset
        let yOffset = scrollView.contentOffset.y;
        // 2. condition 
        if (yOffset < 0) {
            //3. caculate current imageView`s height when scrolling
            let totalOffset = headerViewH - (yOffset);
            //4. caculate  the ratio
            let f = totalOffset / headerViewH;
            //5 .set imageV`s frame again
            imageV.frame =  CGRect(x: screenWidth*(1-f)*0.5, y: yOffset, width: screenWidth*f , height: totalOffset)
        }
    }

}

This tip will enhance user experience a little,i recommand you to use this in your developing apps.
